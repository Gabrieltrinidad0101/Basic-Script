# create VARS
VAR a = expression

# Example
VAR a = 10
VAR b = 9
VAR a = 10 + 10
VAR a = b + 10
VAR a = IF a > b THEN a ELSE b
VAR a = TRUE
VAR a = FALSE

# create IF
IF condition THEN
    expression
END


# Example
IF 10 == 10 THEN
    PRINT("hello world")
END

IF 10 >= 10 THEN
    PRINT("hello world")
END


# create FOR
FOR i = NUMBER TO NUMBER STEP NUMBER THEN
    expression
END

FOR i = NUMBER TO NUMBER THEN
    expression
END

# Example
FOR i = 0 TO 10 STEP 2 THEN
    PRINT(i)
END

FOR i = 0 TO 10 THEN
    PRINT(i)
END

# create WHILE
WHILE condition THEN
    expression
END

# Example
VAR i = 0
WHILE i < 10 THEN
    PRINT(i)
    VAR i = i + 1
END


# create LIST
VAR a = [1,2,3,4]

# add element into list
VAR a = [1,2,3,4] + 1

# remove element into list
VAR a = [1,2,3,4] - 1
